% --- Statistical & Mathematical Notation (APA style: italicized) -------------
% Common statistical symbols — APA 7 requires italic for statistical symbols

% --- Effect Size Reporting (APA style) ----------------------------------------
% Usage: \effectsize{d}{0.45}  →  d = 0.45
\newcommand{\effectsize}[2]{$#1 = #2$}

% --- Confidence Interval Reporting (APA style) --------------------------------
% Usage: \confint{95}{0.35}{0.78}  →  95% CI [0.35, 0.78]
\newcommand{\confint}[3]{#1\%~\CI~[#2,~#3]}

% --- Inline Statistics Reporting Template (APA style) -------------------------
% Usage: \Ftest{2}{47}{4.56}{.02}  →  F(2, 47) = 4.56, p = .02
\newcommand{\Ftest}[4]{\Fstat(#1,~#2)~=~#3,~\pvalue~=~#4}

% Usage: \ttest{47}{2.34}{.03}  →  t(47) = 2.34, p = .03
\newcommand{\ttest}[3]{\tstat(#1)~=~#2,~\pvalue~=~#3}
