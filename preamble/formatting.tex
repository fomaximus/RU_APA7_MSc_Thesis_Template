% --- PDF Metadata & Link Styles (hyperref) -----------------------------------
\hypersetup{%
    pdftitle={\thesisTitle},
    pdfauthor={\thesisAuthor},
    pdfsubject={\thesisProgram{} Thesis},
    pdfkeywords={\thesisKeywords},
    pdfcreator={LuaLaTeX with apa7},
    pdfproducer={LuaLaTeX},
    % Link styling: Remove default red/green boxes and use colored text instead.
    % In a thesis, we want all citations, internal links, and external URLs
    % to be black (clean print standard).
    colorlinks=true,
    linkcolor=black,       % Color of internal links (e.g., sections, figures)
    citecolor=black,       % Color of citation links
    urlcolor=black,        % Color of external URLs (including references)
    hypertexnames=false,   % Prevents duplicate page destination warnings due to Roman/Arabic numbering
    % Alternative: use "hidelinks" to completely hide all link decorations
    % hidelinks
}

% --- Global Justification ---------------------------------------------------
\AtBeginDocument{\justifying}

% --- Front Matter Page Style -------------------------------------------------
% Custom command for creating unnumbered front matter sections
% that still appear in the Table of Contents.
\newcommand{\frontmattersection}[1]{%
    \newpage
    \vspace*{1.8cm}
    \section*{#1}%
    \addcontentsline{toc}{section}{#1}%
}

% --- Table of Contents Depth -------------------------------------------------
% Show sections and subsections in ToC (APA levels 1–2)
\setcounter{tocdepth}{2}
\setcounter{secnumdepth}{0}  % APA style: no section numbering

% --- Double-Sided Page Clearing (RU Thesis requirement) -----------------------
% Forces the next section to start on an odd-numbered (right-hand) page.
\newcommand{\RUcleardoublepage}{%
    \clearpage
    \ifodd\value{page}%
    \else
        \thispagestyle{empty}%
        \hbox{}%
        \clearpage
    \fi
}
