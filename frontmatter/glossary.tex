\frontmattersection{Glossary}

\noindent
\begin{tabular}{@{}p{3.8cm} p{11.7cm}@{}}
    \textbf{dissertation} & A formal discourse, written thesis, or treatise on a subject, usually submitted for an academic degree. \\[0.8\baselineskip]
    \textbf{glossary}     & An alphabetical list of terms in a particular domain of knowledge with the definitions for those terms. \\[0.8\baselineskip]
    \textbf{ohm}          & The SI unit of electrical resistance, expressing the resistance between two points of a conductor. \\
\end{tabular}
